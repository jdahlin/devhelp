enum GtkPrivateFlags

typedef enum
{
  PRIVATE_GTK_USER_STYLE	= 1 <<  0,
  PRIVATE_GTK_REDRAW_PENDING	= 1 <<  1,
  PRIVATE_GTK_RESIZE_PENDING	= 1 <<  2,
  PRIVATE_GTK_RESIZE_NEEDED	= 1 <<  3,

  PRIVATE_GTK_IN_REPARENT       = 1 <<  6,
  PRIVATE_GTK_IS_OFFSCREEN      = 1 <<  7
} GtkPrivateFlags;
